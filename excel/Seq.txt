Sub Seq()
    Dim src_s As String
    src_s = Cells(2, 1).Value
    
    Dim start_i As Long
    start_i = Cells(2, 2).Value
    
    Dim end_i As Long
    end_i = Cells(2, 3).Value
    
    Dim step_i As Long
    If Cells(2, 4).Value = "" Or Cells(2, 4).Value = "0" Then
        step_i = 1
    Else
        step_i = Cells(2, 4).Value
    End If
    
    Dim zero_i As Long
    If Cells(2, 5).Value = "" Or Cells(2, 5).Value = "0" Then
        zero_i = 1
    Else
        zero_i = Cells(2, 5).Value
    End If
    
    Worksheets("Result").Cells.Clear
       
    Dim i As Long
    For start_i = start_i To end_i Step step_i
        Dim r_no As String
        r_no = CStr(start_i)
        Dim zi As Long
        For zi = 1 To zero_i
            If Len(r_no) < zero_i Then
                r_no = "0" & r_no
            End If
        Next
       
        i = i + 1
        Worksheets("Result").Cells(i, 1).Value = src_s & r_no
    Next

        
End Sub
